    Model: példa

    from django.db import models

class MyModel(models.Model):
    myfield = models.CharField(max_length=45)
    myrel = models.ForeignKey(OtherModel)

    class Meta:
      unique_together = [(myfield, myrel)]

    def __unicode__(self):
        return self.myfield

    @models.permalink
    def get_absolute_url(self):
        return ('mymodel', (), {'pk': self.pk})

    ModelManager: példa

    
class VideoManager(models.Manager):
    use_for_related_fields = True

    def recent(self):
        return self.get_query_set()

    def by_user(self, user):
        return self.get_query_set().filter(user=user.get_profile())


class PublishedVideoManager(VideoManager):

    def get_query_set(self):
        return super(PublishedVideoManager, self).get_query_set().filter(project__is_published=True)

class Video(TimeStampedModel):
    ...

    objects = VideoManager()
    published = PublishedVideoManager()
